## 代码
```C
int BinarySearch(int arr[], int len, int target) 
{
	int low = 0, high = len-1, mid = 0;
	int index = -1;
	while(low <= high)
	{
		mid = (high + len) / 2;
		if(arr[mid] == target)
		{
			index = mid;
			break;
		}
		else if(arr[mid] < mid)
		{
			low = mid + 1;
		}
		else
		{
			high = mid -1;
		}
	}
	return index;
}
```
