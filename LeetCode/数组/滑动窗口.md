## 例题
### 209.长度最小的子数组
```cpp
class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums)
    {
        int slow = 0, bestResult = nums.size() + 1, result = 0, sum = 0;
        for (int fast = 0;fast < nums.size();fast++)
        {
            sum += nums[fast];
            while (sum >= target)
            {
                result = fast - slow + 1;
                bestResult = min(result, bestResult);
                sum -= nums[slow];
                slow++;
            }
        }
        if(bestResult==nums.size()+1)
            return 0;
        return bestResult;
    }
};
```
